4. Έστω το παρακάτω πρόγραμμα με δομές επανάληψης

#include <iostream>

using namespace std;

int main()
{
  int a, b, c, d, e;

  b = 0;
  c = 0;
  d = 0;
  e = 0;

  for(a = 0; a < 5; a++)
  {
    b = b + a;
  }

  while(b >= 0)
  {
    c++;
    d = d + 2;
    b--;
  }

  do
  {
    e = e + 4;
    d++;
  }while(d < 10);

  return 0;
}
