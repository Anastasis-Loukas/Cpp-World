1. Έστω το παρακάτω πρόγραμμα

#include <iostream>

using namespace std;

void swap_by_value(int, int);
void swap_by_reference(int&, int&);
void swap_by_pointer(int*, int*);

int main()
{
  int a, b, c, d, e, f;

  a = c = e = 5;
  b = d = f = 10;

  swap_by_value(a, b);
  swap_by_reference(c, d);
  swap_by_pointer(&e, &f);

  return 0;
}

void swap_by_value(int i, int j)
{
  int temp;

  temp = i;
  i = j;
  j = temp;
}

void swap_by_reference(int &i, int &j)
{
  int temp;

  temp = i;
  i = j;
  j = temp;
}

void swap_by_pointer(int *i, int *j)
{
  int temp;

  temp = *i;
  *i = *j;
  *j = temp;
}

2. Αντιστοίχιση

Εξετάζουμε τις συναρτήσεις ως προς τις αλλαγές των μεταβλητών που περνάνε σαν ορίσματα και το αποτύπωμα μνήμης της συνάρτησης

swap_by_value(int, int); a. Διατήρηση των αλλαγών με χρήση λιγότερης μνήμης
swap_by_reference(int&, int&); b. Διατήρηση των αλλαγών με χρήση περισσότερης μνήμης
swap_by_pointer(int*, int*); c. Δεν διατηρούνται οι αλλαγές που πραγματοποιούνται μέσα στην συνάρτηση

using namespace std;

void add(int, int);
void add(long, long);
void add(float, float);
void add(double, double);

int main()
{
  add(2, 4);
  add(2l, 4l);
  add(2.5, 4.0);
  add(2.5f, 4.0f);

  return 0;
}

void add(int i, int j)
{
  cout<<i + j<<endl;

}

void add(long i, long j)
{
  cout<<(i + j) * 2<<endl;

}

void add(float i, float j)
{
  cout<<i + j<<endl;

}

void add(double i, double j)
{
  cout<<(i + j) * 2<<endl;

}
