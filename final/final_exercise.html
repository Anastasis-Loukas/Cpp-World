8. Έστω το παρακάτω πρόγραμμα

#include <iostream>

using namespace std;

void swap_by_value(int, int);
void swap_by_reference(int&, int&);
void swap_by_pointer(int*, int*);

int main()
{
  int a, b, c, d, e, f;

  a = c = e = 5;
  b = d = f = 10;

  swap_by_value(a, b);
  swap_by_reference(c, d);
  swap_by_pointer(&e, &f);

  return 0;
}

void swap_by_value(int i, int j)
{
  int temp;

  temp = i;
  i = j;
  j = temp;
}

void swap_by_reference(int &i, int &j)
{
  int temp;

  temp = i;
  i = j;
  j = temp;
}

void swap_by_pointer(int *i, int *j)
{
  int temp;

  temp = *i;
  *i = *j;
  *j = temp;
}
